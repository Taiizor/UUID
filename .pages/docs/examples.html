Examples

Basic Examples

Generate and Format UUIDs

// Generate a new UUID
var id = new UUID();

// Different string formats
Console.WriteLine(id.ToString());        // Standard format
Console.WriteLine(id.ToBase32String());  // Base32 format
Console.WriteLine(id.ToBase64String());  // Base64 format

// Get raw bytes
byte[] bytes = id.ToByteArray();

This example shows different ways to create and format UUIDs.

Parse UUIDs

// Parse from string
var id1 = UUID.Parse("123e4567-e89b-12d3-a456-426614174000");

// Try parse pattern
if (UUID.TryParse("123e4567-e89b-12d3-a456-426614174000", out var id2))
{
    Console.WriteLine("Successfully parsed: " + id2);
}

// Parse from Base32
var base32 = "6QR8H1RZWM8VDPJ38WSKDD5GR0";
var id3 = UUID.ParseBase32(base32);

Examples of parsing UUIDs from different string formats.

Custom Formatting

// Custom string builder formatting

// Format with prefix

Database Examples

Entity Framework Core

public class User
{
    public UUID Id { get; set; }
    public string Name { get; set; }
}

public class AppDbContext : DbContext
{
    public DbSet Users { get; set; }

    protected override void OnModelCreating(ModelBuilder modelBuilder)
    {
        modelBuilder.Entity()
            .Property(e => e.Id)
            .HasConversion(
                v => v.ToString(),
                v => UUID.Parse(v));
    }
}

Example of using UUIDs with Entity Framework Core.

Dapper

public async Task GetUser(UUID id)
{
    using var connection = new SqlConnection(connectionString);
    return await connection.QuerySingleOrDefaultAsync(
        "SELECT * FROM Users WHERE Id = @Id",
        new { Id = id.ToString() }
    );
}

Web API Examples

ASP.NET Core Controller

public class UserController : ControllerBase
{
    [HttpGet("{id}")]
    public async Task> GetUser(UUID id)
    {
        var user = await _userService.GetUserAsync(id);
        if (user == null)
            return NotFound();
        return user;
    }

    [HttpPost]
    public async Task> CreateUser(User user)
    {
        user.Id = new UUID(); // Generate new UUID
        await _userService.CreateUserAsync(user);
        return CreatedAtAction(nameof(GetUser), new { id = user.Id }, user);
    }
}

Example of using UUIDs in ASP.NET Core Web API.
