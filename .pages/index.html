Quick Start

// Install via NuGet
dotnet add package UUID

// Generate a new UUID
var id = new UUID();

// Convert to string
string str = id.ToString();

// Parse from string
UUID parsed = UUID.Parse("0123456789abcdef0123456789abcdef");
